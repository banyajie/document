###### 一：求数组中第K大的数

// 冒泡排序，第k个出来的数
func FindKThNum(arr []int64, k int) in64 {
    if len(arr) < k {
        return 0
    }    
    
    for i := 0; i < len(arr); i ++ {
        for j := i; j < len(arr)-i-1; j ++ {
            if arr[i] < arr[j+1] {
                arr[i], arr[j+1] = arr[j+1], arr[i] 
            }
        }
        if i == k {
            return arr[i]
        }
    }
    
    return 0
}
